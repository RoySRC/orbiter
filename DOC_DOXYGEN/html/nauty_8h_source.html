222 /*****************************************************************************
223 * *
224 * 16-bit, 32-bit and 64-bit versions can be selected by defining WORDSIZE. *
225 * The largest graph that can be handled has MAXN vertices. *
226 * Both WORDSIZE and MAXN can be defined on the command line. *
227 * WORDSIZE must be 16, 32 or 64; MAXN must be <= NAUTY_INFINITY - 2; *
228 * *
229 * With a very slight loss of efficiency (depending on platform), nauty *
230 * can be compiled to dynamically allocate arrays. Predefine MAXN=0 to *
231 * achieve this effect, which is default behaviour from version 2.0. *
232 * In that case, graphs of size up to NAUTY_INFINITY-2 can be handled *
233 * if the the memory is available. *
234 * *
235 * If only very small graphs need to be processed, use MAXN<=WORDSIZE *
236 * since this causes substantial code optimizations. *
237 * *
238 * Conventions and Assumptions: *
239 * *
240 * A 'setword' is the chunk of memory that is occupied by one part of *
241 * a set. This is assumed to be >= WORDSIZE bits in size. *
242 * *
243 * The rightmost (loworder) WORDSIZE bits of setwords are numbered *
244 * 0..WORDSIZE-1, left to right. It is necessary that the 2^WORDSIZE *
245 * setwords with the other bits zero are totally ordered under <,=,>. *
246 * This needs care on a 1's-complement machine. *
247 * *
248 * The int variables m and n have consistent meanings throughout. *
249 * Graphs have n vertices always, and sets have m setwords always. *
250 * *
251 * A 'set' consists of m contiguous setwords, whose bits are numbered *
252 * 0,1,2,... from left (high-order) to right (low-order), using only *
253 * the rightmost WORDSIZE bits of each setword. It is used to *
254 * represent a subset of {0,1,...,n-1} in the usual way - bit number x *
255 * is 1 iff x is in the subset. Bits numbered n or greater, and *
256 * unnumbered bits, are assumed permanently zero. *
257 * *
258 * A 'graph' consists of n contiguous sets. The i-th set represents *
259 * the vertices adjacent to vertex i, for i = 0,1,...,n-1. *
260 * *
261 * A 'permutation' is an array of n short ints (ints in the case that *
262 * BIGNAUTY is defined) , repesenting a permutation of the set *
263 * {0,1,...,n-1}. The value of the i-th entry is the number to which *
264 * i is mapped. *
265 * *
266 * If g is a graph and p is a permutation, then g^p is the graph in *
267 * which vertex i is adjacent to vertex j iff vertex p[i] is adjacent *
268 * to vertex p[j] in g. *
269 * *
270 * A partition nest is represented by a pair (lab,ptn), where lab and ptn *
271 * are int arrays. The "partition at level x" is the partition whose *
272 * cells are {lab[i],lab[i+1],...,lab[j]}, where [i,j] is a maximal *
273 * subinterval of [0,n-1] such that ptn[k] > x for i <= k < j and *
274 * ptn[j] <= x. The partition at level 0 is given to nauty by the user. *
275 * This is refined for the root of the tree, which has level 1. *
276 * *
277 *****************************************************************************/

817 /* ALLOCS(x,y) should return a pointer (any pointer type) to x*y units of new
818  storage, not necessarily initialised. A "unit" of storage is defined by
819  the sizeof operator. x and y are integer values of type int or larger,
820  but x*y may well be too large for an int. The macro should cast to the
821  correct type for the call. On failure, ALLOCS(x,y) should return a NULL
822  pointer. FREES(p) should free storage previously allocated by ALLOCS,
823  where p is the value that ALLOCS returned. */
824 
825 #ifdef NAUTY_IN_MAGMA
826 #define ALLOCS(x,y) mem_malloc((size_t)(x)*(size_t)(y))
827 #define REALLOCS(p,x) mem_realloc(p,(size_t)(x))
828 #define FREES(p) mem_free(p)
829 #else
830 #define ALLOCS(x,y) malloc((size_t)(x)*(size_t)(y))
831 #define REALLOCS(p,x) realloc(p,(size_t)(x))
832 #define FREES(p) free(p)

835 /* The following macros are used by nauty if MAXN=0. They dynamically
836  allocate arrays of size dependent on m or n. For each array there
837  should be two static variables:
838  type *name;
839  size_t name_sz;
840  "name" will hold a pointer to an allocated array. "name_sz" will hold
841  the size of the allocated array in units of sizeof(type). DYNALLSTAT
842  declares both variables and initialises name_sz=0. DYNALLOC1 and
843  DYNALLOC2 test if there is enough space allocated, and if not free
844  the existing space and allocate a bigger space. The allocated space
845  is not initialised.
846 
847  In the case of DYNALLOC1, the space is allocated using
848  ALLOCS(sz,sizeof(type)).
849  In the case of DYNALLOC2, the space is allocated using
850  ALLOCS(sz1,sz2*sizeof(type)).
851 
852  DYNREALLOC is like DYNALLOC1 except that the old contents are copied
853  into the new space. realloc() is assumed. This is not currently
854  used by nauty or dreadnaut.
855 
856  DYNFREE frees any allocated array and sets name_sz back to 0.
857  CONDYNFREE does the same, but only if name_sz exceeds some limit.
858 */
859 
860 #define DYNALLSTAT(type,name,name_sz) \
861  static type *name; static size_t name_sz=0
862 #define DYNALLOC1(type,name,name_sz,sz,msg) \
863  if ((size_t)(sz) > name_sz) \
864  { if (name_sz) FREES(name); name_sz = (sz); \
865  if ((name=(type*)ALLOCS(sz,sizeof(type))) == NULL) {alloc_error(msg);}}
866 #define DYNALLOC2(type,name,name_sz,sz1,sz2,msg) \
867  if ((size_t)(sz1)*(size_t)(sz2) > name_sz) \
868  { if (name_sz) FREES(name); name_sz = (size_t)(sz1)*(size_t)(sz2); \
869  if ((name=(type*)ALLOCS((sz1),(sz2)*sizeof(type))) == NULL) \
870  {alloc_error(msg);}}
871 #define DYNREALLOC(type,name,name_sz,sz,msg) \
872  {if ((size_t)(sz) > name_sz) \
873  { if ((name = (type*)REALLOCS(name,(sz)*sizeof(type))) == NULL) \
874  {alloc_error(msg);} else name_sz = (sz);}}
875 #define DYNFREE(name,name_sz) if (name_sz) {FREES(name); name_sz = 0;}
876 #define CONDYNFREE(name,name_sz,minsz) \
877  if (name_sz > (size_t)(minsz)) {FREES(name); name_sz = 0;}
